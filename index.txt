    <script>
        // Custom Page Transition / Loader Simulation
        window.addEventListener('load', () => {
            const loader = document.getElementById('loader');
            setTimeout(() => {
                loader.style.opacity = '0';
                setTimeout(() => loader.style.display = 'none', 1000);
            }, 1800);
        });

        // App Navigation Logic
        function navigateTo(state) {
            // Deactivate all active screens
            document.querySelectorAll('.screen').forEach(s => s.classList.remove('active'));
            
            if (state === 'good') {
                document.getElementById('screen-good').classList.add('active');
                triggerLuxuryConfetti();
            } else if (state === 'bad') {
                document.getElementById('screen-bad').classList.add('active');
            } else if (state === 'final') {
                document.getElementById('screen-final').classList.add('active');
                startTypewriter();
            }
        }

        // Celebratory Trigger tailored to palette
        function triggerLuxuryConfetti() {
            const duration = 2.5 * 1000;
            const animationEnd = Date.now() + duration;
            const defaults = { startVelocity: 25, spread: 360, ticks: 60, zIndex: 0 };

            function randomInRange(min, max) {
                return Math.random() * (max - min) + min;
            }

            const interval = setInterval(function() {
                const timeLeft = animationEnd - Date.now();

                if (timeLeft <= 0) {
                    return clearInterval(interval);
                }

                const particleCount = 20 * (timeLeft / duration);
                // Confetti tinted to match editorial color scheme (Burgundy, Gold, Blush)
                confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.1, 0.3), y: Math.random() - 0.2 }, colors: ['#4A1525', '#C5A059', '#E8D7D0'] }));
                confetti(Object.assign({}, defaults, { particleCount, origin: { x: randomInRange(0.7, 0.9), y: Math.random() - 0.2 }, colors: ['#4A1525', '#C5A059', '#E8D7D0'] }));
            }, 250);
        }

        // Editorial Typewriter implementation
        const finalLines = [
            "Bhaad me jaye.",
            "Jo hona tha ho gaya.",
            "Ab bas relax karo, everything will be okay."
        ];

        function startTypewriter() {
            const container = document.getElementById('typewriter');
            container.innerHTML = '';
            let lineIndex = 0;
            let charIndex = 0;

            function type() {
                if (lineIndex < finalLines.length) {
                    if (charIndex === 0) {
                        const p = document.createElement('p');
                        p.style.marginBottom = '12px';
                        if(lineIndex === 2) p.style.color = 'var(--accent-gold)';
                        container.appendChild(p);
                    }
                    
                    const currentParagraph = container.lastChild;
                    currentParagraph.innerHTML += finalLines[lineIndex].charAt(charIndex);
                    charIndex++;

                    if (charIndex < finalLines[lineIndex].length) {
                        setTimeout(type, 40);
                    } else {
                        lineIndex++;
                        charIndex = 0;
                        setTimeout(type, 600); // Elegant pause between thoughts
                    }
                }
            }
            setTimeout(type, 400);
        }

        // Physics Layer: Soft organic falling petals with drag, gravity, and wind sway
        const canvas = document.getElementById('ambient-canvas');
        const ctx = canvas.getContext('2d');

        let width = canvas.width = window.innerWidth;
        let height = canvas.height = window.innerHeight;

        window.addEventListener('resize', () => {
            width = canvas.width = window.innerWidth;
            height = canvas.height = window.innerHeight;
        });

        class Petal {
            constructor() {
                this.x = Math.random() * width;
                this.y = Math.random() * -height - 20;
                this.size = Math.random() * 6 + 4;
                this.gravity = Math.random() * 0.4 + 0.4;
                this.wind = Math.random() * 0.5 - 0.25;
                this.swaySpeed = Math.random() * 0.02 + 0.01;
                this.swayAngle = Math.random() * Math.PI;
                this.opacity = Math.random() * 0.4 + 0.2;
                // Palette distribution for organic materials
                this.color = Math.random() > 0.5 ? '#4A1525' : '#E8D7D0';
            }

            update() {
                this.y += this.gravity;
                this.swayAngle += this.swaySpeed;
                this.x += this.wind + Math.sin(this.swayAngle) * 0.4;

                if (this.y > height + 20) {
                    this.y = -20;
                    this.x = Math.random() * width;
                }
            }

            draw() {
                ctx.save();
                ctx.translate(this.x, this.y);
                ctx.rotate(this.swayAngle * 0.2);
                ctx.beginPath();
                // Draw sophisticated organic particle silhouettes (asymmetrical organic ellipses)
                ctx.ellipse(0, 0, this.size, this.size * 0.6, 0, 0, 2 * Math.PI);
                ctx.fillStyle = this.color;
                ctx.globalAlpha = this.opacity;
                ctx.fill();
                ctx.restore();
            }
        }

        const petalArray = Array.from({ length: 28 }, () => new Petal());

        function animate() {
            ctx.clearRect(0, 0, width, height);
            petalArray.forEach(petal => {
                petal.update();
                petal.draw();
            });
            requestAnimationFrame(animate);
        }
        animate();
    </script>
